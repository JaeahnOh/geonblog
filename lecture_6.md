function checkMenuFloatingLayer() {
	
	var welMenuFloatingLayer = $Element('menuFloatingLayer');
	var welPreviousVolumeBtn = $Element('previousVolumeBtn');
	var welNextVolumeBtn = $Element('nextVolumeBtn');
	var welDetailViewArea = $Element($$.getSingle('.detail_view_area'));
	var welDetailViewContainer = $Element($$.getSingle('.detail_view_container'));
	
	var documentScrollPositionTop = $Document().scrollPosition().top;
	var limitFloatingLayerOffsetTop = welDetailViewArea.offset().top + welDetailViewArea.height() - welMenuFloatingLayer.height();

	var browserExtraPixel = 0;
	var oNavigator = $Agent().navigator();
	
	// IE10 or 크롬에서 발생되는 여백
	if ((oNavigator.ie == true && oNavigator.version == 10) || oNavigator.chrome == true) {
		browserExtraPixel = 1;
	}
	
	if (menuFloatingLayerOffsetTop < documentScrollPositionTop) {
		
		// 플로팅 레이어
		if ($Document().scrollPosition().left > 0) {
			welMenuFloatingLayer.css('position', 'absolute');
			welMenuFloatingLayer.css('top', $Document().scrollPosition().top - welDetailViewArea.offset().top);
			welMenuFloatingLayer.css('left', -1);
		} else {
			welMenuFloatingLayer.css('position', 'fixed');
			welMenuFloatingLayer.css('top', '0');
			welMenuFloatingLayer.css('left', welDetailViewArea.offset().left + browserExtraPixel);
		}
		
		if (limitFloatingLayerOffsetTop <= welMenuFloatingLayer.offset().top) {
			welMenuFloatingLayer.css('position', 'absolute');
			welMenuFloatingLayer.css('top', welDetailViewArea.height() - welMenuFloatingLayer.height());
			welMenuFloatingLayer.css('left', -1);
		} 
		
		// 이전회 버튼
		if (welPreviousVolumeBtn.visible() == true) {
			
			if ($Document().scrollPosition().left > 0) {
				welPreviousVolumeBtn.css('position', 'absolute');
				welPreviousVolumeBtn.css('top', $Document().scrollPosition().top - welDetailViewArea.offset().top + 322);
				welPreviousVolumeBtn.css('left', 0);
			} else {
				welPreviousVolumeBtn.css('position', 'fixed');
				welPreviousVolumeBtn.css('top', volumeBtnOffsetTop - welDetailViewArea.offset().top);
				welPreviousVolumeBtn.css('left', welDetailViewArea.offset().left + browserExtraPixel + 1);
			}
			
			if (welDetailViewContainerOffsetTop + welDetailViewContainerHeight <= welPreviousVolumeBtn.offset().top + volumeBtnHeight) {
				welPreviousVolumeBtn.css('position', 'absolute');
				welPreviousVolumeBtn.css('top', welDetailViewContainerOffsetTop + welDetailViewContainerHeight - welDetailViewArea.offset().top - volumeBtnHeight - 1);
				welPreviousVolumeBtn.css('left', 0);
			} 
		}
		
		// 다음회 버튼
		if (welNextVolumeBtn.visible() == true) {
			if ($Document().scrollPosition().left > 0) {
				welNextVolumeBtn.css('position', 'absolute');
				welNextVolumeBtn.css('top', $Document().scrollPosition().top - welDetailViewArea.offset().top + 322);
				welNextVolumeBtn.css('left', welDetailViewArea.width() - volumeBtnWidth - 2);
			} else {
				welNextVolumeBtn.css('position', 'fixed');
				welNextVolumeBtn.css('top', volumeBtnOffsetTop - welDetailViewArea.offset().top);
				welNextVolumeBtn.css('left',  welDetailViewArea.offset().left + welDetailViewArea.width() - volumeBtnWidth + browserExtraPixel - 1);
			}
			
			if (welDetailViewContainerOffsetTop + welDetailViewContainerHeight <= welNextVolumeBtn.offset().top + volumeBtnHeight) {
				welNextVolumeBtn.css('position', 'absolute');
				welNextVolumeBtn.css('top', welDetailViewContainerOffsetTop + welDetailViewContainerHeight - welDetailViewArea.offset().top - volumeBtnHeight - 1);
				welNextVolumeBtn.css('left', welDetailViewArea.width() - volumeBtnWidth - 2);
			}
		}
		
	} else {
		welMenuFloatingLayer.css('position', 'absolute');
		welMenuFloatingLayer.css('left', '-1px');
		welMenuFloatingLayer.css('top', '-43px');
		
		if (welPreviousVolumeBtn.visible() == true) {
			$Element('previousVolumeBtn').css('position', 'absolute');
			$Element('previousVolumeBtn').css('left', '0px');
			$Element('previousVolumeBtn').css('top', '325px');
		}
		
		if (welNextVolumeBtn.visible() == true) {
			$Element('nextVolumeBtn').css('position', 'absolute');
			$Element('nextVolumeBtn').css('left', 'auto');
			$Element('nextVolumeBtn').css('right', '0px');
			$Element('nextVolumeBtn').css('top', '325px');
		}
	}

}

		fontSize = fontSizeArray[index+1];

		fontSize = fontSizeArray[index-1];

		$("bigFontBtn").disabled = "disabled";
		value = MAX_FONT_SIZE;
	} else if (fontSize <= MIN_FONT_SIZE) {
		$("smallFontBtn").disabled = "disabled";
		value = MIN_FONT_SIZE;

		$("bigFontBtn").removeAttribute("disabled");
		$("smallFontBtn").removeAttribute("disabled");

	for (var i=0; i<fontSizeArray.length; i++) {
		elDetailViewContent.removeClass('ft' + fontSizeArray[i]);

		// 댓글영역을 오버했을때
		if (scrollHeight > welCommentArea.offset().top + welCommentArea.height()) {
			$Element('topButton').css('position', 'absolute');
			$Element('topButton').css('margin-left', welCommentArea.width() + 'px');
			$Element('topButton').css('bottom', '0px');
		} else {
			$Element('topButton').css('position', 'fixed');
			$Element('topButton').css('margin-left', welCommentArea.width() - $Document().scrollPosition().left + 'px');
			$Element('topButton').css('bottom', '50px');
		}
	} else {

		// 스크롤이 작품정보영역 하단 노출 최소높이보다 작다면
		if (scrollHeight < novelInfoAreaTopMinHeight) {

			// 작품정보영역 열려있을때
			if ($Element('novelInfoArea').hasClass('close') == false) {
				$Element('topButton').css('position', 'absolute');
				$Element('topButton').css('margin-left', '760px');
			}
			return;
		}
		
		// 작품정보영역 노출여부로 분기
		if ($Element('novelInfoArea').hasClass('close') == true) {
			$Element('topButton').css('position', 'fixed');
			$Element('topButton').css('margin-left', 870 - $Document().scrollPosition().left + 'px');
		} else {
			$Element('topButton').css('position', 'fixed');
			$Element('topButton').css('margin-left', 760 - $Document().scrollPosition().left + 'px');
		}
	}
}

var oStarScoreLayerManager = null;	// 별점 레이어 Manager
function toggleStarScoreLayer(oEvent, elItem, aParams) {
	oEvent.stopDefault();
	
	// 점검중이면 안내 띄우고 종료
	if (gRosYn == 'Y') {
		alert(gRosAlertMessage);
		return;
	}
	
	// 이미 레이어가 열려있으면 닫기
	if (oStarScoreLayerManager && oStarScoreLayerManager.getVisible() == true) {
		oStarScoreLayerManager.hide();
		oStarScoreLayerManager = null;
		
		return;
	}
	
	// 로그인 체크(로그인 안했으면 로그인 페이지로 이동)
	if (Util.String.isEmpty(gsUserId)) {
		if (confirm('로그인이 필요한 서비스입니다.\n로그인 하시겠습니까?')) {
			novel.login();
		} else {
			return;
		}
	}
	
	// 본인 글에 별점 줄 수 없음
	if (false) {
		alert('본인 글에는 별점을 입력하실 수 없습니다.');
		return;
	}
	
	var bStarScoreCompleted = (aParams[0] == 'true');
	if (bStarScoreCompleted == true) {
		// 이미 참여했으면 이미 참여 레이어 열기
		oStarScoreLayerManager = new jindo.LayerManager("starScoreCompleteLayer", {
			sCheckEvent : "click"
		}).link($("starScoreCompleteLayer")).link($("starScoreBtn"));
	} else {
		// 참여 안했으면 별점 주기 레이어 열기
		oStarScoreLayerManager = new jindo.LayerManager("starScoreLayer", {
			sCheckEvent : "click"
		}).link($("starScoreLayer")).link($("starScoreBtn"));
	}
	oStarScoreLayerManager.attach('hide', function(oCustomEvent) {
		changeStarScore(10);	// 레이어 닫으면 별점 초기화
	});
	oStarScoreLayerManager.show();
}


function hideStarScoreLayer(oEvent, elItem, aParams) {
	oEvent.stopDefault();
	if (oStarScoreLayerManager) {
		oStarScoreLayerManager.hide();
		oStarScoreLayerManager = null;
	}
}


function onChangeStarScore(oEvent, elItem, aParams) {
	oEvent.stopDefault();
	
	var nStarScore = parseInt(aParams[0]);
	changeStarScore(nStarScore);
}

function changeStarScore(nStarScore) {
	if ($Element('starScoreLayer') == null) {
		return;
	}
	$Element('selectedStarScore').html(nStarScore + '<span class="blind">점</span>');

	$ElementList($$('.grade_bon')).removeClass('grade_bon');
	var welStarList = $ElementList($$('._star'));
	for (var i = 0; i < welStarList.length(); i++) {
		if (i < nStarScore) {
			$Element(welStarList.get(i)).addClass('grade_bon');
		}
	}
}


function submitStarScore(oEvent, elItem, aParams) {
	oEvent.stopDefault();
	
	jindo.DomData.remove(elItem, '_submitStarScoreBtn');	// 빠른 클릭으로 요청이 여러번 날라가는 것을 방지
	
	var nNovelId = parseInt(aParams[0]);
	var nVolumeNo = parseInt(aParams[1]);
	var nStarScore = $$('.grade_bon').length;
	
	var oAjax = new $Ajax('/my/volumeStarScoreRegisterJson.nhn', {
		type : 'xhr',
		method : 'get',
		onload : function(oRes) {
			var oJson = oRes.json();
			var result = oJson.processResult;
			var volume = oJson.volume;
			
			// 별점 레이어 닫기
			hideStarScoreLayer(oEvent);
			
			setTimeout(function() {	// 별점 레이어가 닫히고 alert 뜨게 하기 위해서 0.2초 대기
				if (result == 'ALREADY') {
					alert('이미 별점을 입력하셨습니다.');
				} else if (result == 'SUCCESS') {
					
					// 별점 버튼 클릭 이벤트에 파라미터 변경
					jindo.DomData.set($('starScoreBtn'), '_starScoreBtn', [true]);
					
					// 별점 갱신
					$ElementList($$('._starScoreAverage span.grade span')).css('width', volume.starScoreAverage * 10 + '%');
					$ElementList($$('._starScoreAverage em')).text(volume.starScoreAverage);
					$ElementList($$('._starScoreCount strong')).text(Util.Number.changeNumberFormat(parseInt(volume.starScoreCount)));
					
					alert('별점이 등록 되었습니다.');
				} else if (result == 'NOLOGIN') {
					if (confirm('로그인이 필요한 서비스입니다.\n로그인 하시겠습니까?')) {
						naver.login();
					}
				} else if (result == 'INVALID')  {
					alert('잘못된 입력입니다.');
				} else if (result == 'MINE')  {
					alert('본인 글에는 별점을 입력하실 수 없습니다.');
				} else if (result == 'ROS') {
					alert('2013-01-17 18:00~20:00 정기점검중입니다.');
					jindo.DomData.set(elItem, '_submitStarScoreBtn(' + nNovelId + ',' + nVolumeNo + ')');
				} else if (result == 'PENALTY') {
					alert('님은 징계 기간 중이므로 별점기능이 제한됩니다.');
					jindo.DomData.set(elItem, '_submitStarScoreBtn(' + nNovelId + ',' + nVolumeNo + ')');
				}
			}, 200);
		},
		onerror : function(oRes) {
			alert('오류가 발생했습니다. 잠시 후 다시 시도해주세요.');
		},
		ontimeout : function(oRes) {
			alert('처리가 지연되고 있습니다. 다시 시도해주세요.');
		}
	}).request({
		novelId : nNovelId,
		volumeNo : nVolumeNo,
		starScore : nStarScore
	});
	
}
